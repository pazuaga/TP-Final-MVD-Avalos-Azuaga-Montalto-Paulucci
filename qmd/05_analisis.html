Regresión con controles

Estimamos dos modelos de regresión lineal sobre el panel país-año. El primero incluye únicamente la exposición comercial con China como variable explicativa. El segundo incorpora controles por nivel de desarrollo (PBI per cápita), región geográfica y año, para descartar correlaciones espurias y enfrentar la objeción de que el efecto observado refleja desarrollo económico más que exposición a China.

# ── Preparación: armar grupos por mediana ────────────────────
datos_grupos <- dataset_final |>
  filter(!is.na(china_dependence), !is.na(pct_yes)) |>
  mutate(
    grupo = if_else(
      china_dependence >= median(china_dependence, na.rm = TRUE),
      "Alta exposición",
      "Baja exposición"
    ),
    grupo = factor(grupo, levels = c("Alta exposición", "Baja exposición"))
  )

# ── IC sobre la diferencia de medias (t-test de Welch) ───────
# Como el factor tiene Alta primero, t.test calcula Alta - Baja
# Por eso NO invertimos los signos del IC.
test_ic <- t.test(pct_yes ~ grupo, data = datos_grupos)

dif_media <- test_ic$estimate[[1]] - test_ic$estimate[[2]]  # Alta - Baja
ic_inf    <- test_ic$conf.int[1]
ic_sup    <- test_ic$conf.int[2]

df_ic <- tibble(
  estimacion = dif_media,
  ic_inf     = ic_inf,
  ic_sup     = ic_sup,
  etiqueta   = "Alta − Baja\nexposición"
)

# ── Panel A: densidades superpuestas ─────────────────────────
p_densidades <- ggplot(
  datos_grupos,
  aes(x = pct_yes, fill = grupo, color = grupo)
) +

library(tidytext)

# ── 1. Cargar (o descargar) el modelo udpipe ──────────────────
modelo_dir <- here("data/raw")
dir.create(modelo_dir, recursive = TRUE, showWarnings = FALSE)

modelo_existente <- list.files(modelo_dir, pattern = "english.*\\.udpipe$",
                               full.names = TRUE)

if (length(modelo_existente) == 0) {
  cat("Descargando modelo udpipe (~15MB, solo la primera vez)...\n")
  dl <- udpipe_download_model(language = "english", model_dir = modelo_dir)
  modelo_path <- dl$file_model
} else {
  modelo_path <- modelo_existente[1]
}

modelo_udpipe <- udpipe_load_model(modelo_path)

# ── 2. Anotar los textos (lematizar + POS tagging) ────────────
anotacion <- udpipe_annotate(
  modelo_udpipe,
  x = textos_clasificados$descr,
  doc_id = as.character(textos_clasificados$rcid)
) |>
  as.data.frame()

# ── 3. Filtrar a lemas informativos (sustantivos, propios, adjetivos)
lemmas <- anotacion |>
  filter(upos %in% c("NOUN", "PROPN", "ADJ"),
         !is.na(lemma),
         nchar(lemma) > 2) |>
  mutate(lemma = str_to_lower(lemma)) |>
  inner_join(
    textos_clasificados |>
      mutate(doc_id = as.character(rcid)) |>
      select(doc_id, grupo),
    by = "doc_id"
  )

# ── 4. Log odds ratio ponderado por grupo ────────────────────
df_logodds <- lemmas |>
  count(grupo, lemma, sort = TRUE) |>
  bind_log_odds(grupo, lemma, n)

# ── 5. Top 15 lemas por grupo ────────────────────────────────
df_top <- df_logodds |>
